## 三、已实现表结构

### 3.1 系统用户表 (sys_user)

```sql
CREATE TABLE `sys_user` (
  `id` bigint NOT NULL AUTO_INCREMENT,
  `username` varchar(50) NOT NULL COMMENT '登录账号(建议用手机号)',
  `password` varchar(100) NOT NULL COMMENT '登录密码',
  `real_name` varchar(50) NOT NULL COMMENT '真实姓名',
  `phone` varchar(20) DEFAULT NULL,
  `user_type` tinyint NOT NULL COMMENT '1-内部教职工 2-家长',
  `status` tinyint DEFAULT '1' COMMENT '0-禁用 1-正常',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP,
  `update_time` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `is_deleted` tinyint DEFAULT '0' COMMENT '逻辑删除',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_username` (`username`)
);
```

### 3.2 角色表 (sys_role)

```sql
CREATE TABLE `sys_role` (
  `id` bigint NOT NULL AUTO_INCREMENT,
  `role_name` varchar(50) NOT NULL,
  `role_code` varchar(50) NOT NULL,
  `data_scope` tinyint NOT NULL DEFAULT '4' COMMENT '数据范围',
  `description` varchar(255) DEFAULT NULL,
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP,
  `update_time` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `is_deleted` tinyint DEFAULT '0',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_role_code` (`role_code`)
);
```

### 3.3 权限表 (sys_permission)

```sql
CREATE TABLE `sys_permission` (
  `id` bigint NOT NULL AUTO_INCREMENT,
  `parent_id` bigint DEFAULT '0' COMMENT '父级ID',
  `perm_name` varchar(50) NOT NULL COMMENT '菜单或按钮名称',
  `perm_type` tinyint NOT NULL COMMENT '1-目录 2-菜单 3-按钮',
  `route_path` varchar(100) DEFAULT NULL,
  `api_url` varchar(255) DEFAULT NULL,
  `sort_order` int DEFAULT '0',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP,
  `update_time` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `is_deleted` tinyint DEFAULT '0',
  PRIMARY KEY (`id`)
);
```

### 3.4 用户角色关联表 (sys_user_role)

```sql
CREATE TABLE `sys_user_role` (
  `id` bigint NOT NULL AUTO_INCREMENT,
  `user_id` bigint NOT NULL,
  `role_id` bigint NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_user_role` (`user_id`,`role_id`)
);
```

### 3.5 角色权限关联表 (sys_role_permission)

```sql
CREATE TABLE `sys_role_permission` (
  `id` bigint NOT NULL AUTO_INCREMENT,
  `role_id` bigint NOT NULL,
  `permission_id` bigint NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_role_perm` (`role_id`,`permission_id`)
);
```

### 3.6 班级信息表 (busi_class)

```sql
CREATE TABLE `busi_class` (
  `id` bigint NOT NULL AUTO_INCREMENT,
  `class_name` varchar(50) NOT NULL COMMENT '班级名称(如：大一班)',
  `grade` tinyint NOT NULL COMMENT '年级：1-托班 2-小班 3-中班 4-大班',
  `head_teacher_id` bigint DEFAULT NULL COMMENT '班主任ID',
  `capacity` int DEFAULT NULL COMMENT '班级容量',
  `status` tinyint DEFAULT '1' COMMENT '班级状态：1-正常 0-毕业/注销',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP,
  `update_time` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `is_deleted` tinyint DEFAULT '0' COMMENT '逻辑删除',
  PRIMARY KEY (`id`)
);
```

## 四、待实现表结构

### 4.1 幼儿档案表 (busi_student)

```sql
CREATE TABLE `busi_student` (
  `id` bigint NOT NULL AUTO_INCREMENT,
  `class_id` bigint NOT NULL COMMENT '所属班级ID',
  `student_name` varchar(50) NOT NULL COMMENT '幼儿姓名',
  `gender` tinyint DEFAULT '0' COMMENT '性别：1-男 2-女 0-未知',
  `birthday` date DEFAULT NULL COMMENT '出生日期',
  `id_card` varchar(20) DEFAULT NULL COMMENT '身份证号/护照号(敏感信息)',
  `admission_date` date DEFAULT NULL COMMENT '入园日期',
  `status` tinyint DEFAULT '1' COMMENT '在园状态：1-在读 2-毕业 3-退园',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP,
  `update_time` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `is_deleted` tinyint DEFAULT '0' COMMENT '逻辑删除',
  PRIMARY KEY (`id`),
  KEY `idx_class_id` (`class_id`)
);
```

### 4.2 幼儿与家长绑定关系表 (busi_student_parent)

```sql
CREATE TABLE `busi_student_parent` (
  `id` bigint NOT NULL AUTO_INCREMENT,
  `student_id` bigint NOT NULL COMMENT '幼儿ID',
  `parent_user_id` bigint NOT NULL COMMENT '家长用户ID',
  `relation` varchar(20) NOT NULL COMMENT '亲属关系(如：爸爸、妈妈、爷爷)',
  `is_primary` tinyint DEFAULT '0' COMMENT '是否主监护人：0-否 1-是',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_student_parent` (`student_id`, `parent_user_id`)
);
```

### 4.3 教师任教班级分配表 (busi_teacher_class)

```sql
CREATE TABLE `busi_teacher_class` (
  `id` bigint NOT NULL AUTO_INCREMENT,
  `teacher_user_id` bigint NOT NULL COMMENT '教师用户ID',
  `class_id` bigint NOT NULL COMMENT '班级ID',
  `is_head_teacher` tinyint DEFAULT '0' COMMENT '是否为该班班主任：0-否 1-是',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_teacher_class` (`teacher_user_id`, `class_id`)
);
```
